cmake_minimum_required(VERSION 3.10)

project("PROJECTNAME")


#message( SYSTEM: ${CMAKE_SYSTEM_NAME} ) 

add_compile_options(-ggdb)

set(
	NIX_INCLUDE_PATHS
)

set(
	NIX_LIBRARY_PATHS
)

set(
	NIX_LIBRARIES
)

IF(CMAKE_SYSTEM_NAME STREQUAL OpenBSD)
	add_definitions( -DUNIX )
	set(
		INCLUDE_PATHS
			${NIX_INCLUDE_PATHS}
	)
	set(
		LIBRARY_PATHS
			${NIX_LIBRARY_PATHS}
	)
	set(
		LIBRARIES
			${NIX_LIBRARIES}
	)
ENDIF()

IF(CMAKE_SYSTEM_NAME STREQUAL Linux)
	add_definitions( -DLINUX )
	set(
		INCLUDE_PATHS
			${NIX_INCLUDE_PATHS}
	)
	set(
		LIBRARY_PATHS
			${NIX_LIBRARY_PATHS}
	)
	set(
		LIBRARIES
			${NIX_LIBRARIES}
	)
ENDIF()
